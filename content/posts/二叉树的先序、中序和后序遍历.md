# 二叉树的遍历

来说说二叉树的遍历方式：

* **深度优先**：按照二叉树的层序遍历
* **广度优先**：按照二叉树的广度一层一层遍历

## 深度优先遍历

首先先来总结一下**迭代法**的遍历方法：

### 迭代法

#### 先序遍历

先序遍历也叫做前序遍历，是按照根节点，左子树节点，右子树节点的顺序遍历的

```C++
class Solution{
    public:
    	vector<int> preorderTree(TreeNode* root){
            vector<int>res;
            stack<TreeNode*>st;
            
            if(root==nullptr) return res;
            
            // 用一个栈来辅助输出	先压入根节点
            st.push(root);
            while(!st.empty()){
                TreeNode* cur=st.top();
                st.pop();
                // 先输出根节点
                res.push_back(cur->val);
                // 压入右子树节点 左子树节点
                if(cur->right) st.push(cur->right);
                if(cur->left) st.push(cur->left);
            }
            
            return res;
        }
};
```

#### 中序遍历

中序遍历顺序：先左，再中，后右

```C++
class Solution{
    public:
    	vector<int> inorderTree(TreeNode* root){
            vector<int>res;
            stack<int>st;
            
            if(root==nullptr) return res;
            
            TreeNode* cur=root;
            while(cur!=nullptr || !st.empty()){
                if(cur!=nullptr){
                    st.push(cur);
                    cur=cur->left;
                }else{
                    cur=st.top();
                    res.push_back(cur->val);
                    st.pop();
                    
                    cur=cur->right;
                }
            }
            
            return res;
        }
};
```

#### 后序遍历

遍历顺序：先左，再右，后中

**这里采取前序遍历的调整之后的反转**

```C++
class Solution{
    public:
    	vector<int> backTree(TreeNode* root){
            vector<int> res;
            stack<int> st;
            
            st.push(root);
            while(!st.emptu()){
                TreeNode* cur=st.top();
                res.push_back(cur->val);
                st.pop();
                
                if(cur->left) st.push(cur->left);
                if(cur->right) st.push(cur->right);
            }
            
            reverse(res.begin(),res.end());
            return res;
        }
};
```

### 递归法

#### 先序遍历

```C++
class Solution{
    public:
    	void traver(TreeNode* cur,vector<int>& vec){
            if(cur==nullptr) return;
            
            // 先将根节点的值压入
            vec.push_back(cur->val);
            // 压入左子树节点递归
            traver(cur->left,vec);
            // 压入右子树节点递归
            traver(cur->right,vec);
        }
    
    	vector<int> preorderTree(TreeNode* root){
            vector<int> result;
            traver(root,result);
            
            return result;
        }
};
```

#### 中序遍历

```C++
class Solution{
    public:
    	void traver(TreeNode* cur,vector<int>& vec){
            if(cur==nullptr) return;
            
            traver(cur->left,vec);		// 左
            vec.push_back(cur->val);	// 中
            traver(cur->right,vec);		// 右
        }
    	
    
    	vector<int> inorderTree(TreeNode* root){
            vector<int>res;
            traver(root,res);
            
            return res;
        }
};
```

#### 后序遍历

```C++
class Solution{
    public:
    	void traver(TreeNode* cur,vector<int>& vec){
            if(cur==nullptr) return;
            
            traver(cur->left,vec);		// 左
            traver(cur->right,vec);		// 右
            vec.push_back(cur->val);	// 中
        }
    
    	vector<int> backTree(TreeNode* root){
            vector<int> res;
            traver(root,res);
            
            return res;
        }
};
```
